\section*{D. Hướng dẫn cài đặt và chạy dự án}

\begin{enumerate}
  \item \textbf{Clone repository:}
  \begin{lstlisting}[basicstyle=\ttfamily\small, frame=none, numbers=none]
git clone https://github.com/Haidp2005/G13Money.git
cd G13Money
  \end{lstlisting}

  \item \textbf{Cài đặt dependencies:}
  \begin{lstlisting}[basicstyle=\ttfamily\small, frame=none, numbers=none]
flutter pub get
  \end{lstlisting}

  \item \textbf{Tạo file \texttt{flutter.env.json}:}
  \begin{lstlisting}[style=jsonStyle, caption={flutter.env.json}]
{
  "GEMINI_API_KEY": "<your-key>",
  "GEMINI_MODEL": "gemini-1.5-flash",
  "SUPABASE_URL": "<your-url>",
  "SUPABASE_ANON_KEY": "<your-key>"
}
  \end{lstlisting}

  \item \textbf{Chạy ứng dụng:}
  \begin{lstlisting}[basicstyle=\ttfamily\small, frame=none, numbers=none]
flutter run --dart-define-from-file=flutter.env.json
  \end{lstlisting}

  \item \textbf{Build release APK:}
  \begin{lstlisting}[basicstyle=\ttfamily\small, frame=none, numbers=none]
flutter build apk --dart-define-from-file=flutter.env.json
  \end{lstlisting}
\end{enumerate}
